1)CREATE A PROCEDURE TO FIND THE DOCTORS WHO HAVE NOT ATTENDED IN-PATIENTS:

	DELIMITER $$
	CREATE DEFINER=`root`@`localhost` PROCEDURE `DOCTOR_NOTATTENDED_INPATIENT`()
	BEGIN
	select doctor_name from doctor 
	where doctor_name not in
	(select D.doctor_name from doctor D,in_patient P
	where D.doctor_id=P.Doc_ID);
	END$$
	DELIMITER ;

CALLING THE ABOVE PROCEDURE:
	
	CALL `DOCTOR_NOTATTENDED_INPATIENT`();
--------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------

3)CREATE A PROCEDURE TO DISPLAY THE IN-PATIENT’s ID, NAME WHO HAVE GIVEN TEST IN LAB ALONG WITH THEIR CORESSPONDING LAB-NO:

	DELIMITER $$
	CREATE DEFINER=`root`@`localhost` PROCEDURE `lab_inpatient`()
	BEGIN
	SELECT in_patient.P_ID,in_patient.Name,lab.lab_no from in_patient  INNER JOIN  lab ON in_patient.P_ID=lab.P_id;
	END$$
	DELIMITER ;

CALLING THE ABOVE PROCEDURE:

	CALL `lab_inpatient`();
---------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------

5)CREATE A PROCEDURE TO FIND THE AVERAGE BILL AMOUNT :

	DELIMITER $$
	CREATE DEFINER=`root`@`localhost` PROCEDURE `billtotal_avg`()
	BEGIN
	SELECT avg(med_fee+(room_fee*days)) as average_billamount from bill ;
	END$$
	DELIMITER ;

CALLING THE ABOVE PROCEDURE:

	CALL `billtotal_avg`();
---------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------
6)CREATE A PROCEDURE TO FIND THE TOTAL SUM OF BILL AMOUNT :

	DELIMITER $$
	CREATE DEFINER=`root`@`localhost` PROCEDURE `BILL_SUM`()
	BEGIN
	SELECT SUM(med_fee+(room_fee*days)) AS TOTAL_AMOUNT FROM bill;
	END$$
	DELIMITER ;

CALLING THE ABOVE PROCEDURE:
	
	CALL `BILL_SUM`();
